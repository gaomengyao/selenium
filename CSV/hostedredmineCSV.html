hostedredmine
store 1 row
readCSV file:///C:\selenium\CSV\account.csv
while ${row}<4
storeCellValue userName ${row},1
storeCellValue passWord ${row},2
open /login
type id=username ${userName}
type id=password ${passWord}
clickAndWait name=login
storeLocation curURL
gotoIf storedVars["curURL"] == "http://www.hostedredmine.com/login" LOGIN_FAIL
waitForTitle 我的工作台 - Online Project Management :: Free Redmine Hosting
assertTitle 我的工作台 - Online Project Management :: Free Redmine Hosting
verifyText id=loggedas 登录为 ${userName}
clickAndWait link=退出
label LOGIN_FAIL
echo ${userName},${passWord},"LOGIN_FAIL"
storeEval ${row}+1 row
endWhile
